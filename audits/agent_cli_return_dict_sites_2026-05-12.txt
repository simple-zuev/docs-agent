    if not link:

    patterns = [

        r"/spreadsheets/d/([a-zA-Z0-9_-]+)",
        r"/file/d/([a-zA-Z0-9_-]+)",
        r"/folders/([a-zA-Z0-9_-]+)",
    ]
    for pat in patterns:
        m = re.search(pat, link)
        if m:

    if re.fullmatch(r"[a-zA-Z0-9_-]{20,}", link):
        return link

    return None
