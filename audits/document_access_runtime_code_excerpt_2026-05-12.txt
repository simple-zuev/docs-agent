    return re.fullmatch(r"[a-zA-Z0-9_-]{20,}", value) is not None


def normalize_query_for_lookup(query: str) -> str:
    q = query.strip()

    prefixes = [
        "прочитай ",
        "read ",
        "read-doc ",
        "открой ",
        "open ",
        "show ",
        "покажи ",
        "найди ",
        "find ",
        "документ ",
        "doc ",
    ]

    ql = q.lower()
    changed = True
    while changed:
        changed = False
        for prefix in prefixes:
            if ql.startswith(prefix):
                q = q[len(prefix) :].strip()
                ql = q.lower()
                changed = True

    return q

def repo_state_payload() -> dict:
    payload = branch_safety_snapshot(BASE)

    if payload.get("ok"):
        return payload

    return {
        "ok": False,
        "command": "repo-state",
        "error_type": payload.get("error_type", "RepoStateError"),
        "error_message": payload.get(
            "error_message", "Failed to build repo state snapshot."
        ),
        "retryable": bool(payload.get("retryable")),
        "auth_related": False,
        "network_related": False,
        "details": payload,
    }
